import { IconGallery, IconItem, Meta } from '@storybook/blocks';
import React from 'react';
import { socials } from '../../lib/icons';
import { JustaThemeProvider } from '../../lib/providers';

<Meta title="Iconography/Socials" />

# Socials

<JustaThemeProvider>
  <IconGallery>
    {
      Object.keys(socials).map((icons, index) => {
        const iconElement = socials[icons];
        return(
          <IconItem key={index} name={icons} title={icons}>
            {React.createElement(iconElement, { size: 32 })}
          </IconItem>
        )})
    }

  </IconGallery>
</JustaThemeProvider>
